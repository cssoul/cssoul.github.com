confirm的demo

  • 实例1:

    var confirm = new MFL.Confirm({
	tip : '确认要这样做吗',
	onConfirm : function () {
		alert('我确认');
	},
	onCancel : function () {
		alert('我放弃');
	}
});

  • 实例2:

    • new MFL.Confirm({
	tip : '确认要这样做吗',
	onConfirm : function () {
		alert('我确认');
	},
	type : 'alert'
});

  • 实例3:

    var customHtml = '自定义字符串'；
new MFL.Confirm({
	wrap : '.customWrap',
	tip : customHtml,
	onConfirm : function () {
		alert('我确认');
	},
	onCancel : function () {
		alert('我放弃');	
	}
});
